\newlength{\currentparskip}

\makeatletter
\newcommand{\twopager}[1]{%
\setlength{\currentparskip}{\parskip}
	\cleardoubleevenstandardpage
	\adjustbox{trim=0 0 {\textwidth} 0, clip}{%
	\begin{minipage}{2\textwidth}
	\setlength{\parskip}{\currentparskip}
	\addchap{Title}
	\setlength{\columnsep}{\innerMargin}
	\begin{multicols}{2}
	#1
	\end{multicols}
	\end{minipage}
	}%
\cleardoublepage
	\adjustbox{trim={\textwidth} 0 0 0, clip}{%
	\begin{minipage}{2\textwidth}
	\setlength{\parskip}{\currentparskip}
	\addchap{Title}
	\setlength{\columnsep}{\innerMargin}
	\begin{multicols}{2}
	#1
	\end{multicols}
	\end{minipage}
}%
\cleardoublepage
}
\makeatother
